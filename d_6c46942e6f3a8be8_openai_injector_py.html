59 """Get the http headers for AOAI request. 

60 

61 The header, whose name starts with "ms-azure-ai-" or "x-ms-", is used to track the request in AOAI. The 

62 value in this dict will be recorded as telemetry, so please do not put any sensitive information in it. 

63 

64 Returns: 

65 A dictionary of http headers. 

66 """ 

67 # get promptflow info from operation context 

68 operation_context = OperationContext.get_instance() 

69 context_info = operation_context.get_context_dict() 

70 promptflow_info = {k.replace("_", "-"): v for k, v in context_info.items()}

74 

75 # update header with promptflow info 

76 headers.update({f"{PROMPTFLOW_PREFIX}{k}": str(v) if v is not None else "" for k, v in promptflow_info.items()})
