56def reverse_transpose(values: Dict[str, List]) -> List[Dict[str, Any]]: 

57 # Setup a result list same len with values 

58 value_lists = list(values.values()) 

59 _len = len(value_lists[0]) 

60 if any(len(value_list) != _len for value_list in value_lists): 

61 raise Exception(f"Value list of each key must have same length, please check {values!r}.") 

62 result = [] 

63 for i in range(_len): 

64 result.append({}) 

65 for key, vals in values.items(): 

66 for _idx, val in enumerate(vals): 

67 result[_idx][key] = val 

68 return result

145def generate_elapsed_time_messages(func_name: str, start_time: float, interval: int, thread_id: int): 

146 import sys 

147 

148 frames = sys._current_frames() 

149 if thread_id not in frames: 149 ↛ 150line 149 didn't jump to line 150

150 thread_msg = ( 

151 f"thread {thread_id} cannot be found in sys._current_frames, " 

152 + "maybe it has been terminated due to unexpected errors." 

153 ) 

154 else: 

155 frame = frames[thread_id] 

156 stack_msgs = format_user_stacktrace(frame) 

157 stack_msg = "".join(stack_msgs) 

158 thread_msg = f"stacktrace of thread {thread_id}:\n{stack_msg}" 

159 elapse_time = time.perf_counter() - start_time 

160 # Make elapse time a multiple of interval. 

161 elapse_time = round(elapse_time / interval) * interval
