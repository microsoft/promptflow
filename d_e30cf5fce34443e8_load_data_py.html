57def _bfs_dir(dir_path: List[str]) -> Tuple[List[str], List[str]]: 

58 """BFS traverse directory with depth 1, returns files and directories""" 

59 files, dirs = [], [] 

60 for path in dir_path: 

61 for filename in os.listdir(path): 

62 file = Path(path, filename).resolve() 

63 if file.is_file(): 

64 files.append(str(file)) 

65 else: 

66 dirs.append(str(file)) 

67 return files, dirs

70def _handle_dir(dir_path: str, max_rows_count: int, logger: logging.Logger = None) -> pd.DataFrame: 

71 """load data from directory""" 

72 df = pd.DataFrame() 

73 

74 # BFS traverse directory to collect files to load 

75 target_dir = [str(dir_path)] 

76 while len(target_dir) > 0: 

77 files, dirs = _bfs_dir(target_dir) 

78 for file in files: 

79 current_df = _pd_read_file(file, logger=logger) 

80 df = pd.concat([df, current_df]) 

81 length = len(df) 

82 if max_rows_count and length > 0: 

83 if length > max_rows_count: 

84 if logger is None: 

85 logger = module_logger 

86 logger.warning(DATA_EXCEED_LIMITATION, length, max_rows_count) 

87 df = df.head(max_rows_count) 

88 break 

89 # no readable data in current level, dive into next level 

90 target_dir = dirs 

91 return df

106def load_df(local_path: str, logger: logging.Logger = None, max_rows_count: int = None) -> pd.DataFrame: 

107 """load data from local file to df. For the usage of PRS.""" 

108 lp = Path(local_path) 

109 try: 

110 if lp.is_file(): 

111 df = _pd_read_file(local_path, logger=logger) 

112 # honor max_rows_count if it is specified 

113 if max_rows_count and len(df) > max_rows_count: 

114 if logger is None: 

115 logger = module_logger 

116 logger.warning(DATA_EXCEED_LIMITATION, len(df), max_rows_count) 

117 df = df.head(max_rows_count) 

118 else: 

119 df = _handle_dir(local_path, max_rows_count=max_rows_count, logger=logger) 

120 except ValueError as e: 

121 raise InvalidUserData( 

122 message_format="Fail to load invalid data. We support file formats: csv, tsv, json, jsonl, parquet. " 

123 "Please check input data." 

124 ) from e
