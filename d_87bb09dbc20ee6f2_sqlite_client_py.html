23def _get_type(t: type): 

24 if issubclass(t, str): 

25 return Types.string 

26 elif issubclass(t, int): 

27 return Types.integer 

28 elif issubclass(t, datetime): 

29 return Types.datetime 

30 elif issubclass(t, bool): 

31 return Types.boolean 

32 elif issubclass(t, float): 

33 return Types.float 

34 else: 

35 raise ValueError(f"No mapping defined from type {t} to sqlite db type.")

45class SqliteClient: 

46 @classmethod 

47 def create_table_if_not_exists( 

48 cls, db_folder_path: str, db_name: str, class_: type, in_memory: bool = False, timeout_seconds: int = 30 

49 ) -> TableInfo: 

50 """Create db table if not exists. If table exists, check if columns are consistent with input class_. 

51 

52 Note that this method involves disk io, it is not safe to be invoked concurrently. 

53 """ 

54 # Check class_ is decorated by dataclass. 

55 if not dataclasses.is_dataclass(class_): 

56 raise ValueError("Input class type must be decorated with dataclasses.dataclass.") 

57 

58 table_name = class_.__name__ # table name is class name. 

59 

60 col_name_to_field_meta = {field.name: field.metadata for field in fields(class_)} 

61 col_name_to_type = {field.name: field.type for field in fields(class_)} 

62 

63 # Get primary key. 

64 primary_key_columns = [k for k, v in col_name_to_field_meta.items() if v.get(PRIMARY_KEY)] 

65 if len(primary_key_columns) != 1: 

66 raise ValueError(f"One and only one primary key column is allowed! Got {len(primary_key_columns)}") 

67 primary_key = primary_key_columns[0] 

68 

69 # Get index column names. 

70 index_columns = [k for k, v in col_name_to_field_meta.items() if v.get(INDEX)] 

71 

72 if not in_memory and not os.path.exists(db_folder_path): 

73 os.mkdir(db_folder_path) 

74 

75 db_path = ":memory:" if in_memory else f"{db_folder_path}/{db_name}" 

76 # Set check_same_thread=False because different threads might use the same db. 

77 db = dataset.connect( 

78 f"sqlite:///{db_path}?check_same_thread=False", 

79 on_connect_statements=[f"PRAGMA busy_timeout = {timeout_seconds * 1000}"], 

80 ) 

81 

82 # Create or load existing table. 

83 if table_name in db.tables: 

84 table = db.load_table(table_name) 

85 else: 

86 primary_col_type = _get_type(col_name_to_type.get(primary_key)) 

87 table = db.create_table( 

88 table_name=table_name, 

89 primary_id=primary_key, 

90 primary_type=primary_col_type, 

91 ) 

92 

93 # Create columns 

94 for col_name, t in col_name_to_type.items(): 

95 if not table.has_column(col_name): 

96 table.create_column(col_name, _get_type(t)) 

97 

98 # Create index for each index column. 

99 for index_col in index_columns: 

100 if not table.has_index([index_col]): 

101 table.create_index([index_col]) 

102 

103 return TableInfo(db, table, primary_key)

131 @retry(OperationalError, tries=3, delay=0.5, backoff=1) 

132 def get(self, primary_key: str) -> object: 

133 """Query by primary key. Raise NotFoundException if not existed.""" 

134 with self._db: 

135 d = self._table.find_one(**{self._primary_key_column: primary_key}) 

136 if d is None: 

137 raise NotFoundException(f"Entity with primary key {primary_key} not found.") 

138 return self._class(**d) 

139 

140 @retry(OperationalError, tries=3, delay=0.5, backoff=1) 

141 def get_by_field(self, **kwargs) -> List[object]: 

142 """Query by column name. Return empty list if not found any. 

143 For example: table.get_by_field(hash_id='hash-id-value') 

144 """ 

145 with self._db: 

146 return [self._class(**r) for r in self._table.find(**kwargs)]

148 @retry(OperationalError, tries=3, delay=0.5, backoff=1) 

149 def insert(self, entity: object): 

150 """Insert entity.""" 

151 d = asdict(entity) 

152 try: 

153 with self._db: 

154 self._table.insert(d) 

155 except IntegrityError as ex: 

156 raise DuplicatedPrimaryKeyException( 

157 f"Primary key {d.get(self._primary_key_column)} already exists." 

158 ) from ex

161 def upsert(self, entity: object): 

162 """Upsert entity.""" 

163 with self._db:
