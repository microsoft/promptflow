33def generate_prompt_tool(name, content, prompt_only=False, source=None):

35 # Get all the variable name from a jinja template 

36 tool_type = ToolType.PROMPT if prompt_only else ToolType.LLM

38 inputs = get_inputs_for_prompt_template(content) 

39 except Exception as e: 

40 msg = f"Parsing jinja got exception: {e}" 

41 raise JinjaParsingError(msg) from e 

42 

43 if prompt_only: 

44 # Currently this is a hard code for prompt tool 

45 # TODO: Use registration instead of hard code 

46 reserved_keys_to_raise = {"template"} 

47 else: 

48 # Import the tools module to initialize the LLM reserved keys 

49 import promptflow.tools # noqa: F401

52 reserved_keys_to_raise = reserved_keys 

53 

54 for input in inputs: 

55 if input in reserved_keys_to_raise:

58 + " Please change another name." 

59 ) 

60 raise ReservedVariableCannotBeUsed(msg)

62 pattern = f"{COMMENT_START_STRING}(((?!{COMMENT_END_STRING}).)*){COMMENT_END_STRING}" 

63 match_result = re.match(pattern, content) 

64 description = match_result.groups()[0].strip() if match_result else None 

65 # Construct the Tool structure

67 name=name, 

68 description=description, 

69 type=tool_type, 

70 inputs={i: InputDefinition(type=[ValueType.STRING]) for i in inputs},

92def collect_tool_functions_in_module(m): 

93 tools = [] 

94 for _, obj in inspect.getmembers(m): 

95 if is_tool(obj): 

96 # Note that the tool should be in defined in exec but not imported in exec, 

97 # so it should also have the same module with the current function. 

98 if getattr(obj, "__module__", "") != m.__name__: 

99 continue 

100 tools.append(obj) 

101 return tools

104def collect_tool_methods_in_module(m): 

105 tools = [] 

106 for _, obj in inspect.getmembers(m): 

107 if isinstance(obj, type) and issubclass(obj, ToolProvider) and obj.__module__ == m.__name__: 

108 for _, method in inspect.getmembers(obj): 

109 if is_tool(method): 

110 tools.append(method) 

111 return tools

149def load_python_module_from_file(src_file: Path): 

150 # Here we hard code the module name as __pf_main__ since it is invoked as a main script in pf. 

151 src_file = Path(src_file).resolve() # Make sure the path is absolute to align with python import behavior. 

152 spec = importlib.util.spec_from_file_location("__pf_main__", location=src_file) 

153 if spec is None or spec.loader is None: 153 ↛ 154line 153 didn't jump to line 154, because the condition on line 153 was never true

154 raise PythonLoaderNotFound(f"Failed to load python file '{src_file}', please make sure it is a valid .py file.") 

155 m = importlib.util.module_from_spec(spec) 

156 try: 

157 spec.loader.exec_module(m) 

158 except Exception as e: 

159 # TODO: add stacktrace to additional info 

160 raise PythonLoadError(f"Failed to load python module from file '{src_file}', reason: {e}.") from e 

161 return m

213def generate_tool_meta_dict_by_file(path: str, tool_type: ToolType): 

214 """Generate meta for a single tool file, which can be a python file or a jinja template file, 

215 note that if a python file is passed, correct working directory must be set and should be added to sys.path. 

216 """ 

217 tool_type = ToolType(tool_type) 

218 file = Path(path).resolve() 

219 if not file.is_file(): 

220 raise MetaFileNotFound( 

221 message_format="Meta file not found, path '{file_path}', type '{tool_type}'.", 

222 file_path=str(file), 

223 tool_type=tool_type, 

224 ) 

225 try: 

226 content = file.read_text() 

227 except Exception as e: 

228 raise MetaFileReadError( 

229 message_format=("Reading meta file failed, path '{file_path}', type '{tool_type}', exception {exception}."), 

230 file_path=str(file), 

231 tool_type=tool_type, 

232 exception=str(e), 

233 ) from e 

234 

235 name = file.stem 

236 if tool_type == ToolType.PYTHON: 

237 return generate_python_meta_dict(name, content, path) 

238 elif tool_type == ToolType.LLM: 

239 return generate_prompt_meta_dict(name, content, source=path) 

240 elif tool_type == ToolType.PROMPT: 

241 return generate_prompt_meta_dict(name, content, prompt_only=True, source=path) 

242 else: 

243 raise ValueError(f"Unsupported tool type {tool_type}.")

283 def python_load_traceback(self): 

284 """Return the traceback inside user's source code scope. 

285 

286 The traceback inside the promptflow's internal code will be taken off. 

287 """ 

288 exc = self.inner_exception 

289 if exc and exc.__traceback__ is not None: 

290 tb = exc.__traceback__ 

291 # The first three frames are always the code in tool.py who invokes the tool. 

292 # We do not want to dump it to user code's traceback. 

293 tb = get_tb_next(tb, next_cnt=3) 

294 if tb is not None: 

295 te = TracebackException(type(exc), exc, tb) 

296 formatted_tb = "".join(te.format()) 

297 

298 return formatted_tb 

299 return None
