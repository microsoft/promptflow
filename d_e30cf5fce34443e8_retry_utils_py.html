10def retry(exception_to_check: Union[Exception, Tuple[Exception]], tries=4, delay=3, backoff=2, logger=None): 

11 """ 

12 From https://www.saltycrane.com/blog/2009/11/trying-out-retry-decorator-python/ 

13 

14 Retry calling the decorated function using an exponential backoff. 

15 

16 http://www.saltycrane.com/blog/2009/11/trying-out-retry-decorator-python/ 

17 original from: http://wiki.python.org/moin/PythonDecoratorLibrary#Retry 

18 

19 :param exception_to_check: the exception to check. may be a tuple of 

20 exceptions to check 

21 :type exception_to_check: Exception or tuple 

22 :param tries: number of times to try (not retry) before giving up 

23 :type tries: int 

24 :param delay: initial delay between retries in seconds 

25 :type delay: int 

26 :param backoff: backoff multiplier e.g. value of 2 will double the delay 

27 each retry 

28 :type backoff: int 

29 :param logger: log the retry action if specified 

30 :type logger: logging.Logger 

31 """ 

32 

33 def deco_retry(f): 

34 @wraps(f) 

35 def f_retry(*args, **kwargs): 

36 mtries, mdelay = tries, delay 

37 while mtries > 1: 

38 try: 

39 if logger: 

40 logger.info("Running %s, %d more tries to go.", str(f), mtries) 

41 return f(*args, **kwargs) 

42 except exception_to_check: 

43 time.sleep(mdelay) 

44 mtries -= 1 

45 mdelay *= backoff 

46 if logger: 

47 logger.warning("%s, Retrying in %d seconds...", str(exception_to_check), mdelay) 

48 return f(*args, **kwargs) 

49 

50 return f_retry # true decorator 

51 

52 return deco_retry
