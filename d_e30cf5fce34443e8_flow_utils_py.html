11def get_flow_session_id(flow_dir: Union[str, PathLike]): 

12 """ 

13 Get the flow session id from flow. The flow id will be same for same flow in same GIT repo or device. 

14 If the flow locates in GIT repo: 

15 use Repo name + relative path to flow_dir as session id 

16 Otherwise: 

17 use device id + absolute path to flow_dir as session id 

18 :param flow_dir: flow directory 

19 """ 

20 flow_dir = Path(flow_dir).resolve() 

21 if not flow_dir.is_dir(): 

22 flow_dir = flow_dir.parent 

23 try: 

24 from git import Repo 

25 

26 repo = Repo(flow_dir, search_parent_directories=True) 

27 session_id = f"{os.path.basename(repo.working_dir)}/{flow_dir.relative_to(repo.working_dir).as_posix()}" 

28 

29 except Exception: 

30 # failed to get repo, use device id + absolute path to flow_dir as session id 

31 import uuid 

32 

33 device_id = uuid.getnode() 

34 session_id = f"{device_id}/{flow_dir.absolute().as_posix()}" 

35 

36 # hash the value to avoid it gets too long, and it's not user visible. 

37 session_id = hashlib.sha256(session_id.encode()).hexdigest() 

38 return session_id
