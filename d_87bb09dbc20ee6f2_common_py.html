6def reconstruct_metrics_dict(metrics: dict) -> dict: 

7 """ 

8 Input metrics example: { 

9 "accuracy": [ 

10 {"value": 0.9, "variant_id": "variant0"}, 

11 {"value": 0.9, "variant_id": "variant1"}, 

12 {"value": 0.9} 

13 ] 

14 } 

15 

16 Output metrics example: { 

17 "accurcacy": 0.9, 

18 "accuracy.variant0": 0.9, 

19 "accuracy.variant1": 0.9 

20 } 

21 """ 

22 METRIC_VALUE_KEY = "value" 

23 VARIANT_ID_KEY = "variant_id" 

24 

25 new_metrics = dict() 

26 for original_metric_name, value_list in metrics.items(): 

27 for value_dict in value_list: 

28 # if variant id exists, set metric name to "<metric_name>.<variant_id>" 

29 metric_name = original_metric_name 

30 if VARIANT_ID_KEY in value_dict: 

31 metric_name = f"{metric_name}.{value_dict[VARIANT_ID_KEY]}" 

32 value = value_dict[METRIC_VALUE_KEY] 

33 new_metrics.update({metric_name: value}) 

34 

35 return new_metrics
